# Simple 3-bit test program
# Demonstrates MOVL, ADD, and HALT

# 0000 no op
111 001 001 001

# 0001 Move literal 3 into RAM address 5
001 101 011 000

# 0010 Move literal 2 into RAM address 6
001 110 010 000

# 0011 Sub RAM[5] + RAM[6] and store result in RAM[7]
011 111 101 110

# 0100 Add RAM[5] + RAM[6] and store result in RAM[7]
010 111 101 110

# 0101 compare addr 5 (3) and addr 6 (2)
100 010 101 101

# 0110 jgt addr 5 (3) > addr 6 (2)
#110 001 000 000

# 0110 jeq addr 5 (3) > addr 6 (2)
101 000 001 000

# 0111 Move literal 7 into RAM address 5
001 101 111 000

# 1000 Move literal 1 into RAM address 5
001 101 001 000

# 1001 Halt execution
000 000 000 000
